title: "User-Generated Content"
description: "Enable players to upload and share custom content"

The UGC (User-Generated Content) API lets players upload and share content like replays, screenshots, custom levels, or other game-specific files.

## Creating a UGC item

Create a new UGC item to get an upload URL:

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.ugc_item_created.connect(_on_ugc_created)

func create_screenshot_ugc(local_path: String):
    WavedashSDK.create_ugc_item(
        WavedashConstants.UGC_TYPE_SCREENSHOT,
        "My Screenshot",
        "A cool moment in the game",
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        local_path
    )

func _on_ugc_created(response):
    if response.get("success", false):
        var ugc_id = response["data"]["id"]
        print("UGC created: ", ugc_id)
```

```csharp Unity
using Wavedash;

public class UGCExample : MonoBehaviour
{
    public async void CreateScreenshotUGC(string localPath)
    {
        var result = await SDK.CreateUGCItem(
            ugcType: 0,        // 0 = screenshot
            title: "My Screenshot",
            description: "A cool moment in the game",
            visibility: 0,     // 0 = public
            localPath: localPath
        );

        if (result != null)
        {
            Debug.Log($"UGC created: {result["id"]}");
        }
    }
}
```

```javascript JavaScript
const response = await WavedashJS.createUGCItem(
  "replay",           // ugcType
  "My Replay",        // title
  "World 1-1 run",    // description
  "public",           // visibility: "public", "private", or "friends"
  "/path/to/file"     // filePath (optional)
);

if (response.success) {
  const ugcId = response.data;
  console.log(`UGC ID: ${ugcId}`);
}
```
</CodeGroup>

### Visibility options

| Value | Description |
|-------|-------------|
| `public` | Anyone can view |
| `private` | Only the creator can view |
| `friends` | Only friends can view |

## Uploading content

<CodeGroup>
```gdscript Godot
# In Godot, pass the local file path when creating the UGC item
# The SDK handles the upload automatically
func upload_screenshot(local_path: String):
    WavedashSDK.create_ugc_item(
        WavedashConstants.UGC_TYPE_SCREENSHOT,
        "Screenshot",
        "Game screenshot",
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        local_path
    )
```

```csharp Unity
// In Unity, pass the local file path when creating the UGC item
public async void UploadScreenshot(string localPath)
{
    var result = await SDK.CreateUGCItem(
        ugcType: 0,
        title: "Screenshot",
        description: "Game screenshot",
        visibility: 0,
        localPath: localPath
    );
}
```

```javascript JavaScript
// For JS games, write to local storage first, then create UGC with the path
await WavedashJS.writeLocalFile("replays/my-replay.dat", replayData);

const response = await WavedashJS.createUGCItem(
  "replay",
  "My Replay",
  "Description",
  "public",
  "replays/my-replay.dat"  // The SDK handles uploading from this path
);

if (response.success) {
  console.log(`UGC created: ${response.data}`);
}
```
</CodeGroup>

## Downloading content

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.ugc_item_downloaded.connect(_on_ugc_downloaded)

func download_ugc(ugc_id: String, local_path: String):
    WavedashSDK.download_ugc_item(ugc_id, local_path)

func _on_ugc_downloaded(response):
    if response.get("success", false):
        print("UGC downloaded to local path")
```

```csharp Unity
public async void DownloadUGC(string ugcId, string localPath)
{
    var result = await SDK.DownloadUGCItem(ugcId, localPath);

    if (result != null)
    {
        Debug.Log("UGC downloaded to local path");
    }
}
```

```javascript JavaScript
const response = await WavedashJS.downloadUGCItem(
  "ugc_abc123...",       // ugcId
  "downloads/item.dat"   // filePath to download to
);

if (response.success) {
  // File is now in local storage
  const data = await WavedashJS.readLocalFile("downloads/item.dat");
}
```
</CodeGroup>

## Attaching UGC to leaderboard entries

<CodeGroup>
```gdscript Godot
func submit_score_with_replay(leaderboard_id: String, score: int, ugc_id: String):
    WavedashSDK.post_leaderboard_score_with_ugc(leaderboard_id, score, true, ugc_id)
```

```csharp Unity
public async void SubmitScoreWithReplay(string leaderboardId, int score, string ugcId)
{
    var result = await SDK.UploadLeaderboardScore(
        leaderboardId,
        score,
        keepBest: true,
        ugcId: ugcId
    );
}
```

```javascript JavaScript
const scoreResponse = await WavedashJS.uploadLeaderboardScore(
  "speedrun-times",     // leaderboardId

```
</CodeGroup>

## UGC types

| Type | Description |
|------|-------------|
| `replay` | Game replay data |
| `screenshot` | Image capture |
| `level` | Custom level or map |
| `save` | Save game data |
| `skin` | Custom character/item skin |

## Example: Screenshot sharing

<CodeGroup>
```gdscript Godot
func share_screenshot():
    var image = get_viewport().get_texture().get_image()
    var path = "user://screenshots/capture.png"

    DirAccess.make_dir_recursive_absolute("user://screenshots")
    image.save_png(path)

    WavedashSDK.create_ugc_item(
        WavedashConstants.UGC_TYPE_SCREENSHOT,
        "Game Screenshot",
        "Captured at level " + str(current_level),
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        path
    )
```

```csharp Unity
public async void ShareScreenshot()
{
    string path = $"{Application.persistentDataPath}/screenshots/capture.png";
    Directory.CreateDirectory(Path.GetDirectoryName(path));
    ScreenCapture.CaptureScreenshot(path);

    await Task.Yield();

    var result = await SDK.CreateUGCItem(
        ugcType: 0,
        title: "Game Screenshot",
        description: $"Captured at level {currentLevel}",
        visibility: 0,
        localPath: path
    );
}
```

```javascript JavaScript
async function shareScreenshot(imageData) {
  // Save screenshot to local storage first
  await WavedashJS.writeLocalFile("screenshots/capture.png", imageData);

  // Create UGC item with the file path
  const response = await WavedashJS.createUGCItem(
    "screenshot",
    "Game Screenshot",
    `Captured at level ${currentLevel}`,
    "public",
    "screenshots/capture.png"
  );

  if (response.success) {
    return response.data;  // Returns the ugcId
  }
  return null;
}
```
</CodeGroup>
