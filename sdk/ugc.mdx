The UGC (User-Generated Content) API lets players upload and share content like replays, screenshots, custom levels, or other game-specific files.

## Creating a UGC item

Create a new UGC item to get an upload URL:

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.ugc_item_created.connect(_on_ugc_created)

func create_screenshot_ugc(local_path: String):
    WavedashSDK.create_ugc_item(
        WavedashConstants.UGC_TYPE_SCREENSHOT,
        "My Screenshot",
        "A cool moment in the game",
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        local_path
    )

func _on_ugc_created(response):
    if response.get("success", false):
        var ugc_id = response["data"]["id"]
        print("UGC created: ", ugc_id)
```

```csharp Unity
public class UGCExample : MonoBehaviour
{
    public async void CreateScreenshotUGC(string localPath)
    {
        var result = await Wavedash.SDK.CreateUGCItem(
            ugcType: WavedashConstants.UGCItemType.SCREENSHOT,
            title: "My Screenshot",
            description: "A cool moment in the game",
            visibility: WavedashConstants.UGCVisibility.PUBLIC,
            filePath: localPath
        );

        if (result != null)
        {
            Debug.Log($"UGC created: {result["id"]}");
        }
    }
}
```

```javascript JavaScript
const response = await WavedashJS.createUGCItem(
  3,               // ugcType: 3 = game_managed (replays, saves)
  "My Replay",     // title
  "World 1-1 run", // description
  0,               // visibility: 0 = public
  "/path/to/file"  // filePath (optional)
);

if (response.success) {
  const ugcId = response.data;
  console.log(`UGC ID: ${ugcId}`);
}
```
</CodeGroup>

### Visibility options

| Value | Constant | Description |
|-------|----------|-------------|
| `0` | `UGC_VISIBILITY.PUBLIC` | Anyone can view |
| `1` | `UGC_VISIBILITY.FRIENDS_ONLY` | Only friends can view |
| `2` | `UGC_VISIBILITY.PRIVATE` | Only the creator can view |

## Uploading content

<Warning>
**Godot**: File paths must be absolute and start with `OS.get_user_data_dir()`. Do not use `user://` virtual paths.
</Warning>

<CodeGroup>
```gdscript Godot
# In Godot, pass an absolute file path starting with OS.get_user_data_dir()
# The SDK handles the upload automatically
func upload_screenshot(file_name: String):
    var path = OS.get_user_data_dir() + "/" + file_name
    WavedashSDK.create_ugc_item(
        WavedashConstants.UGC_TYPE_SCREENSHOT,
        "Screenshot",
        "Game screenshot",
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        path
    )
```

```csharp Unity
// In Unity, pass the local file path when creating the UGC item
public async void UploadScreenshot(string localPath)
{
    var result = await Wavedash.SDK.CreateUGCItem(
        ugcType: WavedashConstants.UGCItemType.SCREENSHOT,
        title: "Screenshot",
        description: "Game screenshot",
        visibility: WavedashConstants.UGCVisibility.PUBLIC,
        filePath: localPath
    );
}
```

```javascript JavaScript
// For JS games, write to IndexedDB first, then create UGC with the path
await WavedashJS.writeLocalFile("replays/my-replay.dat", replayData);

const response = await WavedashJS.createUGCItem(
  3,                         // ugcType: 3 = game_managed (replays, saves)
  "My Replay",
  "Description",
  0,                         // visibility: 0 = public
  "replays/my-replay.dat"    // The SDK handles uploading from this path
);

if (response.success) {
  console.log(`UGC created: ${response.data}`);
}
```
</CodeGroup>

## Updating a UGC item

Update an existing UGC item's metadata or file:

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.ugc_item_updated.connect(_on_ugc_updated)

func update_ugc_item(ugc_id: String, new_title: String):
    WavedashSDK.update_ugc_item(
        ugc_id,
        new_title,
        "Updated description",
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        null  # Pass file path to update the file, or null to keep existing
    )

func _on_ugc_updated(response):
    if response.get("success", false):
        print("UGC updated successfully")
```

```csharp Unity
public async void UpdateUGCItem(string ugcId, string newTitle)
{
    var result = await Wavedash.SDK.UpdateUGCItem(
        ugcId,
        title: newTitle,
        description: "Updated description",
        visibility: WavedashConstants.UGCVisibility.PUBLIC,
        filePath: null  // Pass file path to update the file, or null to keep existing
    );

    if (result != null)
    {
        Debug.Log("UGC updated successfully");
    }
}
```

```javascript JavaScript
const response = await WavedashJS.updateUGCItem(
  ugcId,              // ugcId to update
  "New Title",        // title (optional)
  "New description",  // description (optional)
  UGC_VISIBILITY_PUBLIC, // visibility (optional)
  null                   // filePath - pass path to update file, or null to keep existing
);

if (response.success) {
  console.log("UGC updated successfully");
}
```
</CodeGroup>

## Downloading content

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.ugc_item_downloaded.connect(_on_ugc_downloaded)

func download_ugc(ugc_id: String, file_name: String):
    # File paths must be absolute and start with OS.get_user_data_dir()
    var path = OS.get_user_data_dir() + "/downloads/" + file_name
    DirAccess.make_dir_recursive_absolute(OS.get_user_data_dir() + "/downloads")
    WavedashSDK.download_ugc_item(ugc_id, path)

func _on_ugc_downloaded(response):
    if response.get("success", false):
        print("UGC downloaded to local path")
```

```csharp Unity
public async void DownloadUGC(string ugcId, string localPath)
{
    var result = await Wavedash.SDK.DownloadUGCItem(ugcId, localPath);

    if (result != null)
    {
        Debug.Log("UGC downloaded to local path");
    }
}
```

```javascript JavaScript
const response = await WavedashJS.downloadUGCItem(
  "ugc_abc123...",       // ugcId
  "downloads/item.dat"   // filePath to download to
);

if (response.success) {
  // File is now in IndexedDB
  const data = await WavedashJS.readLocalFile("downloads/item.dat");
}
```
</CodeGroup>

## UGC types

| Type | Value | Description |
|------|-------|-------------|
| `screenshot` | 0 | Player-captured screenshot |
| `video` | 1 | Player-captured video |
| `community` | 2 | Community-created content (levels, mods, etc.) |
| `game_managed` | 3 | Game-managed content (replays, saves) |
| `other` | 4 | Other UGC types |

<Note>
In Godot, use constants like `WavedashConstants.UGC_TYPE_SCREENSHOT`. In Unity, use `WavedashConstants.UGCItemType.SCREENSHOT`. In JavaScript, use the integer values directly (0, 1, 2, 3, 4).
</Note>

## Example: Screenshot sharing

<CodeGroup>
```gdscript Godot
func share_screenshot():
    var image = get_viewport().get_texture().get_image()
    # Use OS.get_user_data_dir() for absolute paths
    var screenshots_dir = OS.get_user_data_dir() + "/screenshots"
    var path = screenshots_dir + "/capture.png"

    DirAccess.make_dir_recursive_absolute(screenshots_dir)
    image.save_png(path)

    WavedashSDK.create_ugc_item(
        WavedashConstants.UGC_TYPE_SCREENSHOT,
        "Game Screenshot",
        "Captured at level " + str(current_level),
        WavedashConstants.UGC_VISIBILITY_PUBLIC,
        path
    )
```

```csharp Unity
public async void ShareScreenshot()
{
    string path = $"{Application.persistentDataPath}/screenshots/capture.png";
    Directory.CreateDirectory(Path.GetDirectoryName(path));
    ScreenCapture.CaptureScreenshot(path);

    await Task.Yield();

    var result = await Wavedash.SDK.CreateUGCItem(
        ugcType: WavedashConstants.UGCItemType.SCREENSHOT,
        title: "Game Screenshot",
        description: $"Captured at level {currentLevel}",
        visibility: WavedashConstants.UGCVisibility.PUBLIC,
        filePath: path
    );
}
```

```javascript JavaScript
async function shareScreenshot(imageData) {
  // Save screenshot to IndexedDB first
  await WavedashJS.writeLocalFile("screenshots/capture.png", imageData);

  // Create UGC item with the file path
  const response = await WavedashJS.createUGCItem(
    0,                         // ugcType: 0 = screenshot
    "Game Screenshot",
    `Captured at level ${currentLevel}`,
    0,                         // visibility: 0 = public
    "screenshots/capture.png"
  );

  if (response.success) {
    return response.data;  // Returns the ugcId
  }
  return null;
}
```
</CodeGroup>
