When your game runs on Wavedash (either locally with `wvdsh dev` or in production), the SDK is automatically loaded into the page and connected to the Wavedash backend. Your game needs to call `Init` to configure the SDK before using it.

## Calling Init

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.backend_connected.connect(_on_connected)
    WavedashSDK.init({"debug": true, "deferEvents": true})
    WavedashSDK.ready_for_events()

func _on_connected(payload):
    print("SDK is ready!")
    var user_id = WavedashSDK.get_user_id()
    var username = WavedashSDK.get_username()
    print("Playing as: ", username)
```

```csharp Unity
using System.Collections.Generic;

public class GameManager : MonoBehaviour
{
    void Start()
    {
        Wavedash.SDK.Init(new Dictionary<string, object>
        {
            { "debug", true }
        });

        if (Wavedash.SDK.IsReady())
        {
            var user = Wavedash.SDK.GetUser();
            if (user != null)
            {
                Debug.Log($"Playing as: {user["username"]}");
            }
        }
    }
}
```

```javascript JavaScript
if (WavedashJS.isReady()) {
  const response = await WavedashJS.getLeaderboard("high-scores");
}
```
</CodeGroup>

## Configuration options

```typescript
interface WavedashConfig {
  debug?: boolean;                       // Useful for logging and testing — enables verbose output in the browser console
  deferEvents?: boolean;                 // Lets your game finish setup before handling lobby/multiplayer signals
  disableAchievementsAndStats?: boolean; // Turns off achievement tracking in sandbox or demo builds
}
```

All config options are optional.

## Event deferral

By default, the SDK starts delivering signals like `LobbyJoined` and `P2P_CONNECTION_ESTABLISHED` immediately. If your game needs time to load assets, set up scenes, or wire up signal handlers before it's ready to handle those events, use `deferEvents: true`. The SDK queues all signals until your game calls `readyForEvents()`.

<CodeGroup>
```gdscript Godot
func _ready():
    # Initialize with deferEvents so signals are queued during setup
    WavedashSDK.init({"debug": true, "deferEvents": true})

    # Wire up signal handlers
    WavedashSDK.lobby_joined.connect(_on_lobby_joined)
    WavedashSDK.lobby_message.connect(_on_lobby_message)
    WavedashSDK.p2p_connection_established.connect(_on_p2p_connected)

    # Do any other game setup — load scenes, assets, etc.
    _load_game_resources()

    # Now tell the SDK you're ready to receive queued signals
    WavedashSDK.ready_for_events()
```

```csharp Unity
void Awake()
{
    // Initialize with deferEvents so signals are queued during setup
    Wavedash.SDK.Init(new Dictionary<string, object>
    {
        { "debug", true },
        { "deferEvents", true }
    });

    // Wire up event handlers
    Wavedash.SDK.OnLobbyJoined += HandleLobbyJoined;
    Wavedash.SDK.OnLobbyMessage += HandleLobbyMessage;
    Wavedash.SDK.OnP2PConnectionEstablished += HandleP2PConnected;
}

void Start()
{
    // Do any other game setup — load scenes, assets, etc.
    LoadGameResources();

    // Now tell the SDK you're ready to receive queued signals
    Wavedash.SDK.ReadyForEvents();
}
```

```javascript JavaScript
// Initialize with deferEvents so signals are queued during setup
WavedashJS.init({ debug: true, deferEvents: true });

// Set up your signal handler
WavedashJS.setEngineInstance({
  type: "custom",
  SendMessage: (objectName, signalName, value) => {
    // Handle signals...
  }
});

// Do any other game setup — load assets, etc.
await loadGameResources();

// Now tell the SDK you're ready to receive queued signals
WavedashJS.readyForEvents();
```
</CodeGroup>

## Backend connection

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.backend_connected.connect(_on_backend_connected)
    WavedashSDK.backend_disconnected.connect(_on_backend_disconnected)
    WavedashSDK.backend_reconnecting.connect(_on_backend_reconnecting)

func _on_backend_connected(payload):
    print("Connected to Wavedash!")

func _on_backend_disconnected(payload):
    print("Disconnected from Wavedash")

func _on_backend_reconnecting(payload):
    print("Reconnecting...")
```

```csharp Unity
void Awake()
{
    Wavedash.SDK.OnBackendConnected += OnBackendConnected;
    Wavedash.SDK.OnBackendDisconnected += OnBackendDisconnected;
    Wavedash.SDK.OnBackendReconnecting += OnBackendReconnecting;
}

void OnBackendConnected(Dictionary<string, object> data)
{
    Debug.Log("Connected to Wavedash!");
}

void OnBackendDisconnected(Dictionary<string, object> data)
{
    Debug.Log("Disconnected from Wavedash");
}

void OnBackendReconnecting(Dictionary<string, object> data)
{
    Debug.Log("Reconnecting...");
}

void OnDestroy()
{
    Wavedash.SDK.OnBackendConnected -= OnBackendConnected;
    Wavedash.SDK.OnBackendDisconnected -= OnBackendDisconnected;
    Wavedash.SDK.OnBackendReconnecting -= OnBackendReconnecting;
}
```

```javascript JavaScript
// Connection signals are emitted automatically:
// BACKEND_CONNECTED - Successfully connected to backend
// BACKEND_DISCONNECTED - Disconnected from backend
// BACKEND_RECONNECTING - Attempting to reconnect
```
</CodeGroup>

| Signal | Description |
|--------|-------------|
| `BACKEND_CONNECTED` | Successfully connected to backend |
| `BACKEND_DISCONNECTED` | Disconnected from backend |
| `BACKEND_RECONNECTING` | Attempting to reconnect |

## Engine-specific initialization

<CodeGroup>
```gdscript Godot
# Wavedash.gd (Autoload)
extends Node

func _ready():
    WavedashSDK.backend_connected.connect(_on_connected)
    WavedashSDK.backend_disconnected.connect(_on_disconnected)
    WavedashSDK.init({"debug": true, "deferEvents": true})
    WavedashSDK.ready_for_events()

func _on_connected(payload):
    print("Connected to Wavedash!")
    var username = WavedashSDK.get_username()
    print("Playing as: ", username)

func _on_disconnected(payload):
    print("Disconnected from Wavedash")
```

```csharp Unity
using System.Collections.Generic;

public class WavedashManager : MonoBehaviour
{
    void Awake()
    {
        Wavedash.SDK.OnLobbyJoined += HandleLobbyJoined;
        Wavedash.SDK.OnP2PConnectionEstablished += HandleP2PConnected;
    }

    void Start()
    {
        Wavedash.SDK.Init(new Dictionary<string, object>
        {
            { "debug", true }
        });

        if (Wavedash.SDK.IsReady())
        {
            var user = Wavedash.SDK.GetUser();
            if (user != null)
            {
                Debug.Log($"Playing as: {user["username"]}");
            }
        }
    }

    void HandleLobbyJoined(Dictionary<string, object> data)
    {
        Debug.Log("Joined lobby!");
    }

    void HandleP2PConnected(Dictionary<string, object> data)
    {
        Debug.Log("P2P connection established!");
    }
}
```

```javascript JavaScript
function waitForWavedash() {
  if (window.WavedashJS && WavedashJS.isReady()) {
    initGame();
  } else {
    setTimeout(waitForWavedash, 100);
  }
}

waitForWavedash();
```
</CodeGroup>
