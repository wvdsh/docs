---
title: "Lobbies"
description: "Create and manage multiplayer game lobbies"
---

The Lobbies API lets you create multiplayer rooms where players can gather before or during gameplay. Lobbies support real-time messaging, metadata storage, and automatic P2P connection setup.

## Creating a lobby

Create a new lobby and automatically join it:

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.lobby_created.connect(_on_lobby_created)

func create_lobby():
    # Lobby types: PUBLIC, FRIENDS_ONLY, or PRIVATE
    WavedashSDK.create_lobby(WavedashConstants.LOBBY_TYPE_PUBLIC, 4)

func _on_lobby_created(response):
    if response.get("success", false):
        var lobby_id = response["data"]
        print("Created lobby: ", lobby_id)
```

```csharp Unity
public class LobbyExample : MonoBehaviour
{
    public async void CreateLobby()
    {
        var lobbyId = await Wavedash.SDK.CreateLobby(
            lobbyVisibility: WavedashConstants.LobbyVisibility.PUBLIC,
            maxPlayers: 4
        );

        if (!string.IsNullOrEmpty(lobbyId))
        {
            Debug.Log($"Created lobby: {lobbyId}");
        }
    }
}
```

```javascript JavaScript
const response = await WavedashJS.createLobby(
  0,  // visibility: 0 = public, 1 = friends_only, 2 = private
  4   // maxPlayers (optional)
);

if (response.success) {
  const lobbyId = response.data;
  console.log(`Created lobby: ${lobbyId}`);
}
```
</CodeGroup>

### Visibility options

| Value | Constant | Description |
|-------|----------|-------------|
| `0` | `LOBBY_VISIBILITY.PUBLIC` | Anyone can find and join |
| `1` | `LOBBY_VISIBILITY.FRIENDS_ONLY` | Only friends can see and join |
| `2` | `LOBBY_VISIBILITY.PRIVATE` | Only joinable with the lobby ID |

## Listing available lobbies

Find public lobbies to join:

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.got_lobbies.connect(_on_got_lobbies)

func list_lobbies():
    # Note: Godot SDK currently lists all public lobbies
    # friendsOnly filtering is not yet supported
    WavedashSDK.list_available_lobbies()

func _on_got_lobbies(lobbies: Array):
    for lobby in lobbies:
        print(lobby["lobbyId"], ": ", lobby["playerCount"], "/", lobby["maxPlayers"], " players")
```

```csharp Unity
public async void ListLobbies()
{
    // List all public lobbies
    var lobbies = await Wavedash.SDK.ListAvailableLobbies();

    foreach (var lobby in lobbies)
    {
        Debug.Log($"{lobby["lobbyId"]}: {lobby["playerCount"]}/{lobby["maxPlayers"]} players");
    }
}

public async void ListFriendsLobbies()
{
    // List only lobbies with friends
    var lobbies = await Wavedash.SDK.ListAvailableLobbies(friendsOnly: true);

    foreach (var lobby in lobbies)
    {
        Debug.Log($"{lobby["lobbyId"]}: {lobby["playerCount"]}/{lobby["maxPlayers"]} players");
    }
}
```

```javascript JavaScript
// List all public lobbies
const response = await WavedashJS.listAvailableLobbies();

if (response.success) {
  response.data.forEach(lobby => {
    console.log(`${lobby.lobbyId}: ${lobby.playerCount}/${lobby.maxPlayers} players`);
  });
}

// List only lobbies with friends
const friendsResponse = await WavedashJS.listAvailableLobbies(true);
```
</CodeGroup>

### Get the host

<CodeGroup>
```gdscript Godot
func is_host(lobby_id: String) -> bool:
    var host_id = WavedashSDK.get_lobby_host_id(lobby_id)
    var is_host = host_id == WavedashSDK.get_user_id()
    if is_host:
        print("You are the host!")
    return is_host
```

```csharp Unity
public bool IsHost(string lobbyId)
{
    var hostId = Wavedash.SDK.GetLobbyHostId(lobbyId);
    var isHost = hostId == Wavedash.SDK.GetUserId();

    {

    }
    return isHost;
}
```

```javascript JavaScript

```
</CodeGroup>

<Note>

</Note>

### Receiving messages

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.lobby_message.connect(_on_lobby_message)

func _on_lobby_message(payload):
    print(payload["username"], ": ", payload["content"])
```

```csharp Unity
void Awake()
{
    Wavedash.SDK.OnLobbyMessage += HandleLobbyMessage;
}

void HandleLobbyMessage(Dictionary<string, object> data)
{
    Debug.Log($"{data["username"]}: {data["content"]}");
}

void OnDestroy()
{
    Wavedash.SDK.OnLobbyMessage -= HandleLobbyMessage;
}
```

```javascript JavaScript
// Messages are delivered via events
WavedashJS.addEventListener(WavedashJS.Events.LOBBY_MESSAGE, (e) => {
  const message = e.detail;
  console.log(`${message.username}: ${message.content}`);
});
```
</CodeGroup>

## Inviting users

Invite friends or other users to join your lobby:

<CodeGroup>
```gdscript Godot
func _ready():
    WavedashSDK.sent_lobby_invite.connect(_on_invite_sent)
    WavedashSDK.lobby_invite.connect(_on_lobby_invite)

func invite_friend(lobby_id: String, user_id: String):
    WavedashSDK.invite_user_to_lobby(lobby_id, user_id)

func _on_invite_sent(response):
    if response.get("success", false):
        print("Invite sent!")

func _on_lobby_invite(data):
    # Received an invite from another player
    print("Invited to lobby by: ", data["inviterUsername"])
    # Optionally auto-join or show UI
    WavedashSDK.join_lobby(data["lobbyId"])
```

```csharp Unity
void Awake()
{
    Wavedash.SDK.OnLobbyInvite += HandleLobbyInvite;
}

public async void InviteFriend(string lobbyId, string userId)
{
    var success = await Wavedash.SDK.InviteUserToLobby(lobbyId, userId);
    if (success)
    {
        Debug.Log("Invite sent!");
    }
}

void HandleLobbyInvite(Dictionary<string, object> data)
{
    // Received an invite from another player
    Debug.Log($"Invited to lobby by: {data["inviterUsername"]}");
    // Optionally auto-join or show UI
}

void OnDestroy()
{
    Wavedash.SDK.OnLobbyInvite -= HandleLobbyInvite;
}
```

```javascript JavaScript
// Send an invite
const response = await WavedashJS.inviteUserToLobby(lobbyId, friendUserId);

if (response.success) {
  console.log("Invite sent!");
}

// Receive invites via Events.LOBBY_INVITE event
WavedashJS.addEventListener(WavedashJS.Events.LOBBY_INVITE, (e) => {
  const data = e.detail;
  console.log(`Invited to lobby by: ${data.inviterUsername}`);
  // Optionally auto-join or show UI
  WavedashJS.joinLobby(data.lobbyId);
});
```
</CodeGroup>

<Note>
The invited user will receive a `WavedashJS.Events.LOBBY_INVITE` event. Your game should handle this event to show an invitation UI or auto-join the lobby.
</Note>

## Invite links

Generate a shareable invite link for the current lobby:

<CodeGroup>
```csharp Unity
public async void GetInviteLink()
{
    var link = await Wavedash.SDK.GetLobbyInviteLink();
    if (!string.IsNullOrEmpty(link))
    {
        Debug.Log($"Invite link: {link}");
    }
}

public async void GetInviteLinkAndCopy()
{
    // Pass true to also copy the link to the clipboard
    var link = await Wavedash.SDK.GetLobbyInviteLink(copyToClipboard: true);
}
```

```javascript JavaScript
const response = await WavedashJS.getLobbyInviteLink();

if (response.success) {
  console.log(`Invite link: ${response.data}`);
}

// Optionally copy to clipboard
const copied = await WavedashJS.getLobbyInviteLink(true);
```
</CodeGroup>

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| `copyToClipboard` | boolean | No | Copy the link to the clipboard (default: false) |

<Note>
The player must be in a lobby to generate an invite link.
</Note>

## Lobby metadata

Store custom key-value data on the lobby. Only the host can set metadata.

### Set metadata (host only)

<CodeGroup>
```gdscript Godot

    # Only the host can set lobby data
    WavedashSDK.set_lobby_data(lobby_id, "gameMode", "deathmatch")
    WavedashSDK.set_lobby_data(lobby_id, "mapName", "arena_01")
```

```csharp Unity
public void SetGameSettings(string lobbyId)

```

```javascript JavaScript
WavedashJS.setLobbyData(lobbyId, "gameMode", "deathmatch");
```

## Events

| Event | Description | Data |
|-------|-------------|------|
| `WavedashJS.Events.LOBBY_JOINED` | Successfully joined a lobby | `{ success, lobbyId, hostId?, users?, metadata?, message? }` |
| `WavedashJS.Events.LOBBY_USERS_UPDATED` | A user joined or left | `{ userId, username, changeType: "JOINED" \| "LEFT" }` |
| `WavedashJS.Events.LOBBY_MESSAGE` | New message received | `{ username, content, ... }` |
| `WavedashJS.Events.LOBBY_DATA_UPDATED` | Metadata changed | `Record<string, unknown>` (full metadata object) |
| `WavedashJS.Events.LOBBY_KICKED` | Removed from lobby | `{ lobbyId, reason }` |
| `WavedashJS.Events.LOBBY_INVITE` | Received lobby invitation | `{ lobbyId, inviterId, inviterUsername }` |

## P2P integration

When players join a lobby, P2P connections are automatically established. See [P2P Networking](/sdk/p2p).

<CodeGroup>
```gdscript Godot
# After joining a lobby, you can send P2P messages
func broadcast_game_data(data: PackedByteArray):
    WavedashSDK.send_p2p_message("", data, 0, true)
```

```csharp Unity
// After joining a lobby, you can send P2P messages
public void BroadcastGameData(byte[] data)
{
    Wavedash.SDK.BroadcastP2PMessage(data, channel: 0, reliable: true);
}
```

```javascript JavaScript
// After joining a lobby, you can send P2P messages
WavedashJS.broadcastP2PMessage(
  0,                        // channel
  true,                     // reliable
  new Uint8Array([1, 2, 3]) // payload
);
```
</CodeGroup>
